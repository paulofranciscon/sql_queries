-- Consulta max dos backups LOG - Bases de aplicação
SELECT s.name, 
CASE type
 WHEN 'D' THEN 'FULL'
 WHEN 'I' THEN 'DIFF'
 WHEN 'L' THEN 'LOG'
END AS Back, 
MAX(backup_finish_date) as backup_finish_date 
FROM master..sysdatabases s JOIN msdb.dbo.backupset (NOLOCK) ON database_name = s.name
WHERE type = 'L' AND s.name NOT IN ('tempdb','pubs','master','model','msdb','dbtrace','ReportServer','ReportServerTempDB')
GROUP BY s.name, type
order by backup_finish_date;
